  \maketitle

  \section{Including Images in Latex}

      \subsection{ Original Image }
      \includegraphics{image.jpg}

      \subsection{ Change Height Image }
      \includegraphics[height=3cm]{image.jpg}

      \subsection{ Change Width Image }
      \includegraphics[height=5cm]{image.jpg}

      \subsection{ Change Height and width Image }
      \includegraphics[height=8cm, width=5cm]{image.jpg}

      \subsection{ Scale Image }
      \includegraphics[scale=0.5]{image.jpg}

      % Image clipping is specified in this order left, bottom, right and top
      \subsection{ Clipped Image }
      \includegraphics[clip=true, trim=15mm 5mm 7mm 3mm]{image.jpg}

      \subsection{ Rotate Image }
      \includegraphics[angle=60]{image.jpg}


      % Add border to the image
      \fbox{ \includegraphics[angle=60]{image.jpg} }

      % Add border to the image and set its width
      \setlength{\fboxrule}{5pt}
      \fbox{ \includegraphics[angle=60]{image.jpg} }


      % Add border to the image and set its width and also add padding
      \setlength{\fboxrule}{5pt}
      \setlength{\fboxsep}{50pt}
      \fbox{ \includegraphics[angle=60]{image.jpg} }
